\usepackage[french]{babel} % Ajuster french/english au besoin. Règle certains messages comme "Table des matières"

\usepackage[unicode]{hyperref} % Conseillé, ajoute la table des matières aux métadonnées du PDF, et des liens cliquables.

\anneestage{20xx-20xx} % Année du stage

\debutstage{jj/mm/aaaa}

\ville{Ville de l'entreprise}
\pays{pays de l'entreprise}
\maitrestage{Prénom NOM} % Du maître de stage
\maitrestagefonction{Fonction du maître de stage}

\title{Titre du stage en Anglais}
\titre{Titre du stage en Français}

\confidentialfalse % Ou confidentialtrue

\author{Nom de l'étudiant}
